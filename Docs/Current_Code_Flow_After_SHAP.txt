## Step 3: Assign Actions to Parties
For each attack class and each party:

If party is DOMINANT:
    → Gets PRIMARY ACTION from ATTACK_ACTIONS dictionary
    → Example: DDOS detected by Party 3 → Party 3 gets "rate-limit, SYN cookies, ..."

If party is NOT DOMINANT:
    → Gets SECONDARY ACTION (monitoring only)
    → Example: DDOS detected by Party 3 → Party 1 gets "monitor volume/rate patterns and alert"

Code Logic:
for i, name in enumerate(party_names):

    else:
        # Non-dominant parties keep their secondary actions (monitor)

class ActionExecutor:
    def __init__(self):
        self.action_history = []
        self.blocked_ips = set()
    
    def execute_action(self, attack_type, action_string, flow_data, confidence):
        """
        Execute action based on recommendation
        """
        # Parse action string
        actions = [a.strip() for a in action_string.split(',')]
        
        # Execute based on attack type and confidence
        if confidence > 0.9:  # High confidence
            if 'block' in action_string.lower():
                self.block_ip(flow_data.get('source_ip'))
            if 'rate-limit' in action_string.lower():
                self.rate_limit(flow_data.get('source_ip'))
        
        if confidence > 0.7:  # Medium confidence
            self.send_alert(attack_type, action_string, flow_data)
        
        # Always log
        self.log_action(attack_type, action_string, flow_data, confidence)
    
    def block_ip(self, ip_address):
        """Block IP address"""
        if ip_address and ip_address not in self.blocked_ips:
            # TODO: Integrate with firewall API
            print(f"[ACTION] Blocking IP: {ip_address}")
            self.blocked_ips.add(ip_address)
            self.action_history.append({
                'action': 'BLOCK_IP',
                'ip': ip_address,
                'timestamp': time.now()
            })
    
    def rate_limit(self, ip_address):
        """Rate limit IP address"""
        # TODO: Integrate with load balancer API
        print(f"[ACTION] Rate limiting IP: {ip_address}")
        self.action_history.append({
            'action': 'RATE_LIMIT',
            'ip': ip_address,
            'timestamp': time.now()
        })
    
    def send_alert(self, attack_type, actions, flow_data):
        """Send alert to security team"""
        # TODO: Integrate with SIEM/alerting system
        print(f"[ALERT] {attack_type} detected. Actions: {actions}")
        self.action_history.append({
            'action': 'ALERT',
            'attack_type': attack_type,
            'timestamp': time.now()
        })
    
    def log_action(self, attack_type, action_string, flow_data, confidence):
        """Log all actions for audit"""
        log_entry = {
            'timestamp': time.now(),
            'attack_type': attack_type,
            'actions': action_string,
            'confidence': confidence,
            'flow_id': flow_data.get('flow_id'),
            'source_ip': flow_data.get('source_ip')
        }
        # TODO: Write to audit log
        print(f"[LOG] {log_entry}")
